# Python program to calculate the first 10,000 terms of the series:
# 1 - 1/3 + 1/5 - 1/7 + ..., then multiply the total by 4.

import math

# Calculate the sum of the first 10,000 terms of the series
n_terms = 10000
total = 0.0
for n in range(n_terms):
    term = (-1) ** n / (2 * n + 1)
    total += term

# Multiply by 4 as per the series
result = total * 4

print(f'Sum of first {n_terms} terms: {total}')
print(f'Result after multiplying by 4: {result}')
print(f'Approximation of pi: {result}')
print(f'Difference from math.pi: {abs(result - math.pi)}')

# Output:
# Sum of first 10000 terms: 0.7853731633975086
# Result after multiplying by 4: 3.1414926535900345
# Approximation of pi: 3.1414926535900345
# Difference from math.pi: 9.99999997586265e-05
